$$ -\alpha f = \beta \frac{d^2 f}{dt^2}, \,\, f(0)=0,\,\, \dot{f}=\gamma $$
$$\dot{f}_{i+1} = \dot{f}_i + \ddot{f_i}h$$$$f_{i+1} = f_i + \dot{f}h$$

def f3(x):

N3 = 3000
if N3%3 !=0:
    N3 = N3 - N3%3  # N3 has to be divisible by 3!

yvals = f3(xvals)

def f4(x):
    return np.sin(x)/x

N4 = 3000
xvals = np.linspace(-np.pi, np.pi, N4)
yvals = f4(xvals)
print(sum(xvals==0))  # make sure we don't get a zero
plt.plot(xvals, yvals)
plt.show()

0

(Rands within the curve)/(Total number of rands) = (Area of the curve)/(Total area)

(Area of the curve) = (Total area) x (Rands within)/(Total Rands)

x0 = xvals[0]  # range for x

curve = f4(xrands)  # the curve for the random x values

yrands_in_curve = yrands<curve  # filter for values within the curve

y_in = yrands[yrands_in_curve]  # values within the curve (for the plot)
